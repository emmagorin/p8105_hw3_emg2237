``` r
instacart %>% 
  ungroup() %>% 
  mutate(aisle = factor(aisle)) %>% 
  group_by(aisle) %>% 
  mutate(count_aisle = n()) %>% 
  filter(count_aisle > 10000) %>% 
  select(aisle, count_aisle, everything()) %>% 
  ggplot(aes(x = fct_rev(fct_infreq(aisle)))) +
  geom_bar() +
  coord_flip() +

    y = "Count of items ordered",
    x = "Aisle name"
  )
```
